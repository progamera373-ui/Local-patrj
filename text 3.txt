local selectedPath = nil
local autoWalkEnabled = false
local walkSpeed = 100
local currentTween = nil
local singleRun = false

local EXACT_NAMES_TO_DESTROY = {
    "MovingWall1", "MovingWall2", "MovingWall3", "MovingWall4", "MovingWall5",
    "MovingWall6", "MovingWall7", "MovingWall8", "MovingWall9",
    "MovingWalls",

    "Twomp",
    "Twomps",
    "Ventilateurs",
    "FanEffects",
}

local PROTECTED_NAMES = {
    ["Bridge1"] = true,
}

local function isExactNameTargeted(name)
    if PROTECTED_NAMES[name] then return false end
    for _, target in ipairs(EXACT_NAMES_TO_DESTROY) do
        if name == target then return true end
    end
    return false
end

local destroyedSet = setmetatable({}, {__mode = "k"})

local function safeDestroy(instance)
    if destroyedSet[instance] then return end
    destroyedSet[instance] = true
    pcall(function()
        instance:Destroy()
    end)
end

local function removeObstacles()
    local removedCount = 0
    local allDescendants = Workspace:GetDescendants()
    for _, descendant in ipairs(allDescendants) do
        if descendant.Parent and isExactNameTargeted(descendant.Name) then
            safeDestroy(descendant)
            removedCount = removedCount + 1
        end
    end
    print("[NexaHub] REMOVE OBSTACLES: " .. removedCount .. " instances removed.")
    return removedCount
end

local speed200Enabled = false
local speed300Enabled = false

mainTab:AddSwitch("Walk Speed 200", function(enabled)
    speed200Enabled = enabled
    if enabled then
        speed300Enabled = false
        walkSpeed = 200
    else
        walkSpeed = 100
    end
end)

mainTab:AddSwitch("Walk Speed 300", function(enabled)
    speed300Enabled = enabled
    if enabled then
        speed200Enabled = false
        walkSpeed = 300
    else
        walkSpeed = 100
    end
end)

mainTab:AddSwitch("Auto Wim", function(enabled)
    autoWalkEnabled = enabled
    singleRun = false
    if enabled then
        removeObstacles()
        print("[NexaHub] Auto Wim iniciado")
    else
        if currentTween then
            currentTween:Cancel()
            currentTween = nil
        end
        print("[NexaHub] Auto Wim detenido")
    end
end)

mainTab:AddButton("REMOVE OBSTACLES", function()
    removeObstacles()
end)

mainTab:AddButton("TP 40M wim no sirve", function()
    selectedPath = paths["TP40M"]
    singleRun = true
    autoWalkEnabled = true
    local character = game.Players.LocalPlayer.Character
    if character and character:FindFirstChild("HumanoidRootPart") then
        character.HumanoidRootPart.CFrame = CFrame.new(paths["TP40M"][1])
    end
    removeObstacles()
    print("[NexaHub] TP 40M wim iniciado (una sola vez)")
end)

task.spawn(function()
    while task.wait() do
        if autoWalkEnabled and selectedPath then
            local character = game.Players.LocalPlayer.Character
            if not character or not character:FindFirstChild("HumanoidRootPart") then
                print("[NexaHub] Esperando al personaje...")
                task.wait(1)
                continue
            end
            local hrp = character.HumanoidRootPart

            for _, part in ipairs(character:GetDescendants()) do
                if part:IsA("BasePart") and part.CanCollide then
                    part.CanCollide = false
                end
            end

            for i = 1, #selectedPath do
                if not autoWalkEnabled then break end
                local targetPos = selectedPath[i]

                while autoWalkEnabled and (hrp.Position - targetPos).Magnitude > 2 do
                    hrp.Velocity = Vector3.zero
                    hrp.RotVelocity = Vector3.zero

                    local distance = (hrp.Position - targetPos).Magnitude
                    local tweenTime = distance / walkSpeed
                    local tweenInfo = TweenInfo.new(tweenTime, Enum.EasingStyle.Linear)

                    currentTween = game:GetService("TweenService"):Create(hrp, tweenInfo, {CFrame = CFrame.new(targetPos)})
                    currentTween:Play()

                    print("[NexaHub] Moviendo al nodo " .. i .. "/" .. #selectedPath)

                    currentTween.Completed:Wait()
                    currentTween = nil
                    task.wait()
                end
            end

            if autoWalkEnabled then
                if character and character:FindFirstChild("Humanoid") then
                    character.Humanoid.Jump = true
                end

                local vim = game:GetService("VirtualInputManager")
                pcall(function()
                    vim:SendKeyEvent(true, Enum.KeyCode.One, false, game)
                    task.wait(1)
                    vim:SendKeyEvent(false, Enum.KeyCode.One, false, game)
                end)

                if singleRun then
                    autoWalkEnabled = false
                    singleRun = false
                    print("[NexaHub] TP 40M wim completado. Deteniendo.")
                else
                    print("[NexaHub] Recorrido completado. Repitiendo...")
                end
                task.wait(1)
            end
        end
    end
end)
